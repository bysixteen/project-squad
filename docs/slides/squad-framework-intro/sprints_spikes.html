// The Spike Qualification Test
function isValidSpike(task) {
  if (task.questions > 1) {
    return false; // Split it
  }
  if (!task.hasBoundedScope) {
    return false; // Too vague
  }
  return true;
}
Output: A clear "Yes/No" or "Recommendation" with evidence.
